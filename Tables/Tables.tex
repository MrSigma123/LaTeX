	% Here we will use multirow environment
	
	\begin{table}[h!]
		\begin{center}
			\caption{Multirow table.}
			\label{tab:table5}
			\begin{tabular}{l|S|r}
				\textbf{Value 1} & \textbf{Value 2} & \textbf{Value 3}\\
				$\alpha$ & $\beta$ & $\gamma$ \\
				\hline
				\multirow{2}{*}{12} & 1110.1 & a\\ % <-- Combining 2  rows with arbitrary with (*) and content 12
				& 10.1 & b\\ % <-- Content of first column omitted.
				\hline
				3 & 23.113231 & c\\
				4 &  25.113231 & d\\
			\end{tabular}
		\end{center}
	\end{table}
	
	\begin{table}[h!]
