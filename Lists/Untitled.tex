\documentclass{article}

\usepackage{enumitem} %necessary package for modified ordering

\begin{document}

% unordered lists
\section{Unordered list}
\begin{itemize}
	\item One
	\item Two
	\item Three
\end{itemize}

% ordered lists
\section{Ordered list}
\begin{enumerate}
    \item One
    \item Two
    \item Three
\end{enumerate}

% nested lists
\section{Nested lists}
\begin{enumerate}
	 \item One
    	\begin{enumerate}
        		\item Two
      		  \item Three
       		 \item Four
   	 \end{enumerate}
    	\item Five
    	\item Six
\end{enumerate}

% modifying bullets or numbering
\section{Modified bullets}

% From bullet to dash \item[--] or \item[$-$]
% From bullet to asterisk \item[$\ast$]
%Use any math character \item[$\CHARACTER$]

% for unordered lists
\begin{itemize}
    \item[--] Dash
    \item[$-$] Dash
    \item[$\ast$] Asterisk
\end{itemize}

% modified numbering
\section{Modified numbering}
Roman, Arabic and Alphabetical numbering examples
\newline

%Roman numbers \begin{enumerate}[label=(\roman*)]
% Arabic numbers \begin{enumerate}[label=\arabic*)]
% Alphabetical \begin{enumerate}[label=\alph*)]

% using Roman numbering
Roman numbering
\begin{enumerate}[label=(\roman*)]
    \item One
    \item Two
    \item Three
\end{enumerate}

% using Arabic numbering
Arabic numbering
\begin{enumerate}[label=(\arabic*)]
    \item One
    \item Two
    \item Three
\end{enumerate}

% using Alphabetical numbering
Alphabetical numbering
\begin{enumerate}[label=(\alph*)]
    \item One
    \item Two
    \item Three
\end{enumerate}
