	\item One
	\item Two
	\item Three
\end{itemize}

% ordered lists
\section{Ordered list}
\begin{enumerate}
    \item One
    \item Two
    \item Three
\end{enumerate}

% nested lists
\section{Nested lists}
\begin{enumerate}
	 \item One
    	\begin{enumerate}
        		\item Two
      		  \item Three
       		 \item Four
   	 \end{enumerate}
    	\item Five
    	\item Six
\end{enumerate}

% modifying bullets or numbering
\section{Modified bullets}

    \item[$-$] Dash
    \item[$\ast$] Asterisk
\end{itemize}
